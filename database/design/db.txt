Table group {
  id integer [primary key, ref: <> user.id]
  name varchar
  created_at timestamp
  updated_at timestamp
}

Table user {
  id integer [primary key]
  username varchar
  firstname varchar
  lastname varchar
  email varchar
  phone_number varchar
  profile_image varchar
  created_at timestamp
  updated_at timestamp
}

Table transaction {
  id integer [primary key]
  group_id fk [ref: - group.id]
  creator_id fk [ref: > user.id]
  last_updated_user_id fk [ref: > user.id]
  amount varchar
  description text
  transaction_type enum [note: "payment, settlement"]
  created_at timestamp
  updated_at timestamp
}

Table transaction_user {
  id integer [primary key]
  transaction_id fk [ref: > transaction.id]
  user_id fk [ref: - user.id]
  amount varchar
  transaction_user_type enum [note: "payer, payee"]
}
